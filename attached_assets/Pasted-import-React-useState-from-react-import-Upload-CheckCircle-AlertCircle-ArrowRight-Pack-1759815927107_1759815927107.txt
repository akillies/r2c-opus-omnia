                  if (swap === true && item.alternative) {

                  } else if (typeof swap === 'string') {
                    const comparable = item.comparables?.find(c => c.id === swap);
                    if (comparable) {
                      displayItem = {...item, ...comparable};
                    }
                  }
                  
                  return (
                    <div key={item.id} className="flex items-center justify-between py-2 border-b border-slate-200 last:border-0">
                      <div className="flex-1">
                        <div className="text-sm font-medium text-slate-900">{displayItem.name}</div>
                        <div className="text-xs text-slate-600">{item.quantity} Ã— ${displayItem.price}</div>
                      </div>
                      <div className="text-sm font-semibold text-slate-900">${(item.quantity * displayItem.price).toFixed(2)}</div>
                    </div>
                  );
                })}
                <div className="pt-4 border-t-2 border-slate-300 flex items-center justify-between">
                  <span className="font-bold text-lg text-slate-900">Total</span>
                  <span className="text-2xl font-bold text-blue-600">${calculateTotal().toFixed(2)}</span>
                </div>
              </div>

              {/* Contract & Compliance */}
              <div className="bg-blue-50 border border-blue-200 rounded-lg p-4 space-y-2">
                <div className="flex items-center gap-2 text-sm font-medium text-blue-900">
                  <CheckCircle className="w-4 h-4 text-blue-600" />
                  All items under STATE-EDU-ABC contract
                </div>
                <div className="flex items-center gap-2 text-sm font-medium text-blue-900">
                  <CheckCircle className="w-4 h-4 text-blue-600" />
                  Budget approved: ${budgetData.total.toLocaleString()}
                </div>
                <div className="flex items-center gap-2 text-sm font-medium text-blue-900">
                  <CheckCircle className="w-4 h-4 text-blue-600" />
                  Expected delivery: 3-5 business days
                </div>
              </div>

              <div className="flex gap-3">
                <button 
                  onClick={() => alert('Cart locked to OPUS! ðŸŽ‰\n\nYour order has been processed and an audit pack is being generated.')}
                  className="flex-1 bg-red-600 hover:bg-red-700 text-white py-4 rounded-lg font-bold transition-all flex items-center justify-center gap-2 shadow-sm"
                >
                  <CheckCircle className="w-5 h-5" />
                  Lock Cart to OPUS
                </button>
                <button className="px-6 py-4 bg-slate-100 hover:bg-slate-200 text-slate-900 rounded-lg font-semibold transition-all border border-slate-300">
                  Download Audit
                </button>
              </div>

              <div className="text-center">
                <button 
                  onClick={() => {setStep(0); setUploadComplete(false); setSelectedSwaps({});}}
                  className="text-sm text-slate-500 hover:text-slate-900 transition-colors font-medium"
                >
                  Start New Flow
                </button>
              </div>
            </div>
          )}
        </div>
      </div>

      {/* Swap Drawer */}
      {showSwapDrawer && currentSwapItem && (
        <div className="fixed inset-0 bg-black/40 backdrop-blur-sm flex items-center justify-end z-50" onClick={() => setShowSwapDrawer(false)}>
          <div className="bg-white h-full w-full max-w-2xl shadow-2xl overflow-y-auto" onClick={e => e.stopPropagation()}>
            <div className="sticky top-0 bg-white border-b border-slate-200 p-6 flex items-center justify-between shadow-sm">
              <div>
                <h3 className="text-xl font-bold text-slate-900">AI-Recommended Alternative</h3>
                <p className="text-sm text-slate-600">Our system suggests this optimized swap</p>
              </div>
              <button onClick={() => setShowSwapDrawer(false)} className="p-2 hover:bg-slate-100 rounded-lg transition-colors">
                <X className="w-5 h-5 text-slate-600" />
              </button>
            </div>

            <div className="p-6 space-y-6">
              {/* Current Selection */}
              <div className="bg-slate-50 border border-slate-200 rounded-lg p-6">
                <div className="text-xs font-semibold text-slate-600 mb-3 uppercase tracking-wider">Current Selection</div>
                <div className="font-semibold text-lg text-slate-900 mb-2">{currentSwapItem.name}</div>
                <div className="grid grid-cols-2 gap-4 mt-4">
                  <div>
                    <div className="text-xs text-slate-600 mb-1">Price per {currentSwapItem.uom}</div>
                    <div className="text-2xl font-bold text-blue-600">${currentSwapItem.price}</div>
                  </div>
                  <div>
                    <div className="text-xs text-slate-600 mb-1">Total Cost</div>
                    <div className="text-2xl font-bold text-slate-900">${(currentSwapItem.price * currentSwapItem.quantity).toFixed(2)}</div>
                  </div>
                </div>
                <div className="mt-4 pt-4 border-t border-slate-200 flex items-center gap-2 text-sm text-slate-700">
                  <Package className="w-4 h-4" />
                  {currentSwapItem.contract}
                </div>
              </div>

              {/* Alternative */}
              <div className="bg-green-50 border border-green-200 rounded-lg p-6">
                <div className="text-xs font-semibold text-green-700 mb-3 uppercase tracking-wider flex items-center gap-2">
                  <Zap className="w-4 h-4" />
                  Recommended Alternative
                </div>
                <div className="font-semibold text-lg text-slate-900 mb-2">{currentSwapItem.alternative.name}</div>
                
                <div className="bg-green-100 border border-green-200 rounded p-3 mb-4">
                  <div className="text-sm font-semibold text-green-900 mb-1">Why this swap?</div>
                  <div className="text-sm text-green-800">{currentSwapItem.alternative.reason}</div>
                </div>

                <div className="grid grid-cols-2 gap-4 mb-4">
                  <div>
                    <div className="text-xs text-slate-600 mb-1">Price per {currentSwapItem.alternative.uom}</div>
                    <div className="text-2xl font-bold text-green-700">${currentSwapItem.alternative.price}</div>
                  </div>
                  <div>
                    <div className="text-xs text-slate-600 mb-1">Total Cost</div>
                    <div className="text-2xl font-bold text-slate-900">${(currentSwapItem.alternative.price * currentSwapItem.quantity).toFixed(2)}</div>
                  </div>
                </div>

                {currentSwapItem.alternative.savings !== 0 && (
                  <div className={`flex items-center gap-2 p-3 rounded ${
                    currentSwapItem.alternative.savings > 0 
                      ? 'bg-green-100 text-green-800 border border-green-200' 
                      : 'bg-yellow-100 text-yellow-800 border border-yellow-200'
                  }`}>
                    <TrendingDown className="w-5 h-5" />
                    <span className="font-semibold">
                      {currentSwapItem.alternative.savings > 0 ? 'Saves' : 'Adds'} ${Math.abs(currentSwapItem.alternative.savings).toFixed(2)}
                    </span>
                  </div>
                )}
              </div>

              <div className="flex gap-3">
                <button 
                  onClick={() => acceptSwap(currentSwapItem.id)}
                  className="flex-1 bg-green-600 hover:bg-green-700 text-white py-4 rounded-lg font-bold transition-all flex items-center justify-center gap-2"
                >
                  <CheckCircle className="w-5 h-5" />
                  Accept AI Swap
                </button>
                <button 
                  onClick={() => setShowSwapDrawer(false)}
                  className="px-8 py-4 bg-slate-100 hover:bg-slate-200 text-slate-900 rounded-lg font-semibold transition-all border border-slate-300"
                >
                  Keep Original
                </button>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Browse Alternatives Drawer */}
      {showBrowseDrawer && browseItem && (
        <div className="fixed inset-0 bg-black/40 backdrop-blur-sm flex items-center justify-end z-50" onClick={() => setShowBrowseDrawer(false)}>
          <div className="bg-white h-full w-full max-w-3xl shadow-2xl overflow-y-auto" onClick={e => e.stopPropagation()}>
            <div className="sticky top-0 bg-white border-b border-slate-200 p-6 flex items-center justify-between shadow-sm">
              <div>
                <h3 className="text-xl font-bold text-slate-900">Browse Comparable Alternatives</h3>
                <p className="text-sm text-slate-600">Select a different product that meets your requirements</p>
              </div>
              <button onClick={() => setShowBrowseDrawer(false)} className="p-2 hover:bg-slate-100 rounded-lg transition-colors">
                <X className="w-5 h-5 text-slate-600" />
              </button>
            </div>

            <div className="p-6 space-y-4">
              {/* Current Selection */}
              <div className="bg-blue-50 border border-blue-200 rounded-lg p-5">
                <div className="text-xs font-semibold text-blue-700 mb-3 uppercase tracking-wider flex items-center gap-2">
                  <CheckCircle className="w-4 h-4" />
                  Current Selection
                </div>
                <div className="flex items-start justify-between">
                  <div className="flex-1">
                    <div className="font-semibold text-lg text-slate-900 mb-1">{browseItem.name}</div>
                    <div className="text-sm text-slate-700">{browseItem.supplier}</div>
                  </div>
                  <div className="text-right">
                    <div className="text-2xl font-bold text-blue-600">${browseItem.price}</div>
                    <div className="text-xs text-slate-600">per {browseItem.uom}</div>
                  </div>
                </div>
                <div className="mt-3 pt-3 border-t border-blue-200 text-sm text-slate-700 font-medium">
                  Total: ${(browseItem.price * browseItem.quantity).toFixed(2)} ({browseItem.quantity} units)
                </div>
              </div>

              {/* Comparable Alternatives */}
              <div className="space-y-3">
                <div className="text-sm font-semibold text-slate-900 flex items-center gap-2">
                  <Package className="w-4 h-4" />
                  {browseItem.comparables?.length || 0} Comparable Products
                </div>
                
                {browseItem.comparables?.map((comp) => {
                  const priceDiff = (browseItem.price - comp.price) * browseItem.quantity;
                  const isSelected = selectedSwaps[browseItem.id] === comp.id;
                  
                  return (
                    <div 
                      key={comp.id} 
                      className={`rounded-lg p-5 border-2 transition-all cursor-pointer ${
                        isSelected 
                          ? 'bg-green-50 border-green-500' 
                          : 'bg-white border-slate-200 hover:border-blue-300 hover:shadow-sm'
                      }`}
                      onClick={() => selectManualSwap(browseItem.id, comp.id)}
                    >
                      <div className="flex items-start justify-between mb-3">
                        <div className="flex-1">
                          <div className="font-semibold text-slate-900 mb-1 flex items-center gap-2">
                            {comp.name}
                            {isSelected && <CheckCircle className="w-4 h-4 text-green-600" />}
                          </div>
                          <div className="text-sm text-slate-700 mb-2">{comp.supplier}</div>
                          <div className="text-xs text-slate-600 bg-slate-100 rounded px-2 py-1 inline-block border border-slate-200">
                            {comp.specs}
                          </div>
                        </div>
                        <div className="text-right ml-4">
                          <div className="text-xl font-bold text-slate-900">${comp.price}</div>
                          <div className="text-xs text-slate-600">per {comp.uom}</div>
                        </div>
                      </div>

                      <div className="flex items-center justify-between pt-3 border-t border-slate-200">
                        <div className="flex items-center gap-3 text-xs">
                          <div className="flex items-center gap-1 bg-blue-100 text-blue-700 px-2 py-1 rounded font-medium">
                            <Package className="w-3 h-3" />
                            {comp.contract}
                          </div>
                          <div className={`flex items-center gap-1 px-2 py-1 rounded font-medium ${
                            comp.availability === 'In Stock' 
                              ? 'bg-green-100 text-green-700' 
                              : 'bg-yellow-100 text-yellow-700'
                          }`}>
                            {comp.availability}
                          </div>
                        </div>
                        <div className={`text-sm font-bold ${
                          priceDiff > 0 ? 'text-green-700' : priceDiff < 0 ? 'text-red-600' : 'text-slate-600'
                        }`}>
                          {priceDiff > 0 ? 'Saves' : priceDiff < 0 ? 'Adds' : 'Same'} ${Math.abs(priceDiff).toFixed(2)}
                        </div>
                      </div>
                    </div>
                  );
                })}
              </div>

              <div className="pt-4">
                <button 
                  onClick={() => setShowBrowseDrawer(false)}
                  className="w-full bg-blue-600 hover:bg-blue-700 text-white py-4 rounded-lg font-bold transition-all"
                >
                  {selectedSwaps[browseItem.id] && typeof selectedSwaps[browseItem.id] === 'string' ? 'Confirm Selection' : 'Close'}
                </button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default MagicFlowPrototype;
